[//]: # (## 버블정렬)

[//]: # (+ 인접한 두 원소를 비교하는 정렬)

[//]: # (+ 시간복잡도 : O&#40;n<sup>2</sup>&#41; / 이미 정렬된 경우 O&#40;N&#41;)

[//]: # (+ 장점)

[//]: # (    + 인접한 값만 계속해서 비교하는 방식으로 구현이 쉬우며 직관적이다.)

[//]: # (    + n개의 원소에 대해 n개의 메모리를 사용하기에 데이터를 하나씩 정밀 비교가 가능하다.)

[//]: # (+ 단점)

[//]: # (    + 최선이든 최약이든 O&#40;n<sup>2</sup>&#41; 시간복잡도를 가진다.)

[//]: # (    + n개 원소에 대해 n개의 메모리를 사용하기에 원소개수가 많아지면 비교횟수가 많아져 성능이 저하된다.)

[//]: # (```java)

[//]: # (// boolean swapped = false;)

[//]: # (for &#40;int i=1; i<size; i++&#41; {)

[//]: # (    for &#40;int j=0; j<size-i; j++&#41; {)

[//]: # (        if &#40;a[j] > a[j+1]&#41; {)

[//]: # (            swap&#40;a, j, j+1&#41;;)

[//]: # (            // swapped = true;)

[//]: # (        })

[//]: # (    })

[//]: # (    // if &#40;swapped == false&#41; {)

[//]: # (        break;)

[//]: # (    })

[//]: # (})

[//]: # (## 퀵 정렬)

[//]: # (+ 하나의 리스트를 pivot을 기준으로 두 개의 부분리스트로 나누어 하나는 pivot보다 작은 값들의 부분리스트, 다른하나는 pivot보다 큰 값들의 부분리스트로 정렬한 다음, 각 부분리스트에 대해 다시 위 처럼 재귀적으로 수행하여 정렬하는 방법)

[//]: # (+ 분할 정복 알고리즘을 기반으로 하는 정렬)

[//]: # (+ 시간복잡도 : O&#40;nlogn&#41; / 최악일때 O&#40;n<sup>2</sup>&#41; => pivot 값을 중간으로 설정하여 개선 가능)

[//]: # (+ 단점)

[//]: # (    + 기존값에 따라 시간복잡도가 크게 달라진다&#40;안정성이 없다&#41;)

[//]: # ()
[//]: # (```java)

[//]: # (private static void quickSort&#40;int[] a, int lo, int hi&#41; {)

[//]: # (    int pivot = partition&#40;a, lo, hi&#41;;)

[//]: # (    if &#40;start < pivot-1&#41; quickSort&#40;a, lo, pivot-1&#41;;)

[//]: # (    if &#40;end > pivot&#41; quickSort&#40;a, pivot, hi&#41;;)

[//]: # (})

[//]: # ()
[//]: # (private static int partition&#40;int[] a, int left, int right&#41; {)

[//]: # (    int pivot = a[&#40;left + right&#41; / 2];)

[//]: # ()
[//]: # (    while &#40;start <= end&#41; {)

[//]: # (        while &#40;arr[left] < pivot&#41; left++;)

[//]: # (        while &#40;arr[right] > pivot&#41; right--;)

[//]: # (        if &#40;left <= right&#41; {)

[//]: # (            swap&#40;a, left, right&#41;;)

[//]: # (            left++;)

[//]: # (            right--;)

[//]: # (    })

[//]: # (    return left;)

[//]: # (})

[//]: # (```)

[//]: # (## 합병정렬)

[//]: # (+ 데이터를 계속 나눈 후, 더 이상 분할이 불가능 할 때 합병하면서 정렬)

[//]: # (+ 시간복잡도 : O&#40;nlogn&#41;)

[//]: # (+ 장점)

[//]: # (    + 항상 O&#40;nlogn&#41; 시간복잡도를 가진다.)

[//]: # (+ 단점)

[//]: # (    + 추가적인 메모리가 필요하다.)

[//]: # ()
[//]: # (```java)

[//]: # (public static void mergeSort&#40;int start, int end&#41; {)

[//]: # (    if &#40;start < end&#41; {)

[//]: # (        int mid = &#40;start + end&#41; / 2;)

[//]: # (        mergeSort&#40;start, mid&#41;;)

[//]: # (        mergeSort&#40;mid+1, end&#41;;)

[//]: # ()
[//]: # (        int sub1Start = start;)

[//]: # (        int sub2Start = mid + 1;)

[//]: # (        int idx = start;)

[//]: # ()
[//]: # (        while &#40;sub1Start <= mid || sub2Start <= end&#41; {)

[//]: # (            if &#40;sub2Start > end || &#40;sub1Start <= mid && src[sub1Start] < src[sub2Start]&#41;&#41; tmp[idx++] = src[sub1Start++];)

[//]: # (            else tmp[idx++] = src[sub2Start++];)

[//]: # (        })

[//]: # (        )
[//]: # (        for &#40;int i=start; i<=end; i++&#41; {)

[//]: # (            src[i] = tmp[i];)

[//]: # (        })

[//]: # (    })

[//]: # (})

[//]: # (```)

[//]: # (## 삽입정렬)

[//]: # (+ 앞에서부터 최소값으로 채우는 정렬)

[//]: # (+ 시간복잡도 : 최선 O&#40;n&#41; / 최악 O&#40;n<sup>2</sup>&#41;)

[//]: # ()
[//]: # (```java)

[//]: # (for &#40;int i=1; i<size; i++&#41; {)

[//]: # (    int idx = i;)

[//]: # (    int tmp = arr[i];)

[//]: # (    )
[//]: # (    while &#40; &#40;idx > 0&#41; && &#40;arr[idx-1] > tmp&#41; &#41; {)

[//]: # (        arr[idx] = arr[idx-1];)

[//]: # (        idx--;)

[//]: # (    })

[//]: # (    arr[idx] = tmp;    )

[//]: # (})

[//]: # (## 선택정렬)

[//]: # (+ 가장 작은 원소를 맨 앞으로 보내기)

[//]: # (+ 시간복잡도 : O&#40;n<sup>2</sup>&#41;)

[//]: # (+ 장점)

[//]: # (    + 비교횟수가 많으나 교환횟수가 적기에, 교환이 많이 이루어져야 하는 상태에서 효율적으로 사용할 수 있다.)

[//]: # (    + 내림차순이 정렬되어 있는 자료를 오름차순으로 재정렬할 때 적합하다.)

[//]: # (+ 단점)

[//]: # (    + 정렬을 위한 비교횟수가 많다.)

[//]: # (    + 이미 정렬된 상태에서 재정렬할 때 최악의 처리 속도를 보인다.)

[//]: # ()
[//]: # (```java)

[//]: # (for &#40;int i=0; i<size-1; i++&#41; {)

[//]: # (    int min_idx = i;)

[//]: # (    )
[//]: # (    for &#40;int j=i+1; j<size; j++&#41; {)

[//]: # (        if &#40;a[j] < a[min_idx]&#41; {)

[//]: # (            min_idx = j;)

[//]: # (        })

[//]: # (    })

[//]: # (    swap&#40;a, min_idx, i&#41;;)

[//]: # (})

[//]: # (```)

[//]: # (    + O&#40;nlogn&#41; 정렬 방법 중 가장 효율적인 정렬방법이다.)

[//]: # (## 기수정렬)

[//]: # (+ 낮은 자리 수 부터 비교하여 정렬해 간다는 것을 기본 개념으로 하는 정렬 알고리즘)

[//]: # (+ 시간복잡도 : O&#40;n&#41;)

[//]: # (+ 장점)

[//]: # (    + 자리수를 비교해 정렬하기 때문에 문자열도 정렬이 가능하다.)

[//]: # (    + 안정적으로 정렬이 가능하다.)

[//]: # (+ 단점)

[//]: # (    + 자릿수가 없는 것은 정렬이 불가능하다.)

[//]: # (    + 추가적인 메모리 공간이 필요하다)

[//]: # ()
[//]: # (```java)

[//]: # (import java.util.Arrays;)

[//]: # (public class radix {)

[//]: # (        // 배열에서 최대 값을 얻기위한 메서드)

[//]: # (        static int getMax&#40;int[] data&#41; {)

[//]: # (            int max = data[0];)

[//]: # (            for &#40;int i=1; i<data.length; i++&#41; {)

[//]: # (                if &#40;data[i] > max&#41; {)

[//]: # (                    max = data[i];)

[//]: # (                })

[//]: # (            })

[//]: # (            return max;)

[//]: # (        })

[//]: # (        )
[//]: # (        // exp 변수의 수에 따라 숫자를 정렬)

[//]: # (        // exp는 자릿수를 나타내는 변수)

[//]: # (        static void countSort&#40;int[] data, int exp&#41; {)

[//]: # (            int[] output = new int[data.length];)

[//]: # (            int[] count = new int[10];)

[//]: # (            Arrays.fill&#40;count, 0&#41;;)

[//]: # (            )
[//]: # (            // count 배열을 자릿수 값에 따라 count)

[//]: # (            for &#40;int i=0; i<data.length; i++&#41; {)

[//]: # (                count[&#40;data[i]/exp&#41;%10]++;)

[//]: # (            })

[//]: # (            )
[//]: # (            // count 배열 누적)

[//]: # (            for &#40;int i=1; i<10; i++&#41; {)

[//]: # (                count[i] += count[i-1];)

[//]: # (            })

[//]: # (            )
[//]: # (            // 거꾸로 for문 도는 이유 : 중간에 정렬된 숫자들을 output 배열 뒤에 둬 유지시키기 위해)

[//]: # (            for &#40;int i=data.length-1; i>=0; i--&#41; {)

[//]: # (                output[count[&#40;data[i]/exp&#41;%10]-1] = data[i];)

[//]: # (                count[&#40;data[i]/exp&#41;%10]--;)

[//]: # (            })

[//]: # (            )
[//]: # (            for &#40;int i=0; i<data.length; i++&#41; {)

[//]: # (                data[i] = output[i];)

[//]: # (            })

[//]: # (        })

[//]: # (})
